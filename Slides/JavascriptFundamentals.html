Prototype

                    
var Persona = function() {
  this.puedeHablar = true;
};

Persona.prototype.saludar = function() {
  if (this.puedeHablar) {
    console.log('Hola, mi nombre es ' + this.nombre);
  }
};

var Empleado = function(nombre, titulo) {
  Persona.call(this);
  this.nombre = nombre;
  this.titulo = titulo;
};

Empleado.prototype = Object.create(Persona.prototype);
Empleado.prototype.constructor = Empleado;

Empleado.prototype.saludar = function() {
  if (this.puedeHablar) {
    console.log('Hola, yo soy ' + this.nombre + ' el ' + this.titulo);
  }
};

var pedro = new Empleado('Pedro', 'Constructor');

pedro.saludar();

Nunca usar variables globales

                        
PROHIBIDO

                        
var a = 5;//puede ser modificado por cualquier otro script que se cargue despues que este

Nunca usar variables globales

                        
RECOMENDADO

                        
(function(){
    var a = 5;
})(); //solo puede ser modificado dentro de este closure

Nunca usar variables globales

                        
PROHIBIDO

                        
(function(){
    a = 5;
})(); //como la variable no tiene var se define a nivel global

Nunca usar variables globales

                        
RECOMENDADO

                        
(function(){
    'use strict';
     var a = 5;
})(); //strict mode nos obliga a declarar la variable local por que sino obtenemos un error al usar una variable no declarada
- 'use strict'; es solo reconocido al inicio del bloque
